workflows:
  ios-build:
    name: iOS Build
    max_build_duration: 60

    environment:
      node: 20
      xcode: latest

    scripts:
      - name: Install dependencies
        script: |
          npm install

      - name: Install CocoaPods
        script: |
          cd ios
          pod install
          cd ..

      - name: Build iOS archive
        script: |
          xcodebuild \
            -workspace ios/TestApp.xcworkspace \
            -scheme TestApp \
            -configuration Release \
            -sdk iphoneos \
            -archivePath $CM_BUILD_DIR/TestApp.xcarchive archive

      - name: Export IPA
        script: |
          xcodebuild -exportArchive \
            -archivePath $CM_BUILD_DIR/TestApp.xcarchive \
            -exportOptionsPlist ios/exportOptions.plist \
            -exportPath $CM_BUILD_DIR/build

    artifacts:
      - $CM_BUILD_DIR/build/*.ipa
